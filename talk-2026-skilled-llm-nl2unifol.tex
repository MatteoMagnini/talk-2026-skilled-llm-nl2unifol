\begin{frame}{Contribution}

\begin{block}{Main contributions of our work}
    \begin{itemize}
        \item Introduce \textbf{NL2UNIFOL}, a framework that turns NL sentences into a \alert{uniform} FOL theory
        \item Propose a \alert{symbol clustering} method that safely merges predicate and constant names
        \item Combine \emph{semantic} and \emph{syntactic} similarity under logical constraints
        \item Empirically study the quality of the merges on a \alert{nutrition} subset of \alert{MALLS}~\cite{DBLP:conf/acl/YangXPSF24}
    \end{itemize}
\end{block}

\section[Background]{Motivation \& Context}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}{Motivation: Norms from Heterogeneous Stakeholders}

\begin{block}{Norms need to be \emph{machine-usable}}
\begin{itemize}
    \item Legal and ethical norms are mostly expressed in NL
    \item For automated decision-making we need them in a \alert{logical formalism}:
    \begin{itemize}
        \item classical logical reasoners (e.g., first-order theorem provers)
        \item argumentation frameworks and other AI reasoning systems
    \end{itemize}
    \item Manual formalisation is \emph{time-consuming}, \emph{error-prone}, and requires experts in both the domain and logic
\end{itemize}
\end{block}

\begin{block}{Sources of normative text}
\begin{itemize}
    \item Official regulations (e.g., GDPR, MiFID II), guidelines, institutional policies
    \item Individual or group descriptions of moral values and principles
\end{itemize}
\end{block}

\begin{block}{Vocabulary misalignment}
\begin{itemize}
    \item Different words for the same entity: \emph{customer} vs \emph{client}
    \item Different phrasings for the same relation: \emph{is employed by} vs \emph{works for}
    \item Typos, singular/plural, synonyms, hypernyms, paraphrases
\end{itemize}
\end{block}

\begin{block}{Consequence}
\begin{itemize}
    \item If each sentence is translated independently into FOL with its own vocabulary,
          classical reasoners and argumentation frameworks \emph{cannot easily relate} different norms
          and derive coherent conclusions over the whole corpus.
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Limit of Existing NL$\rightarrow$FOL Approaches}

\begin{block}{Recent progress}
\begin{itemize}
    \item LLM-based methods for NL$\rightarrow$FOL translation
    \item Benchmarks (e.g., MALLS\footnote{\url{https://huggingface.co/datasets/yuan-yang/MALLS-v0}}, FOLIO~\cite{DBLP:conf/emnlp/HanS0QRZCPQBSWS24}) and evaluation metrics
    \item Strong results on \alert{single-sentence} translation
\end{itemize}
\end{block}

\begin{block}{Missing piece}
\begin{itemize}
    \item Existing works focus on isolated sentences
    \item Predicate naming, constants, and variables are decided \alert{locally}
    \item No explicit mechanism for:
    \begin{itemize}
        \item symbol reuse
        \item predicate alignment
        \item theory-level coherence across a corpus
    \end{itemize}
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Example NL$\rightarrow$FOL Pairs from MALLS (Nutrition)}

\tiny
\begin{table}[t]
\centering
\begin{tabular}{p{0.46\textwidth} p{0.46\textwidth}}
\textbf{Natural language} & \textbf{FOL formula}\\[0.2em]
\hline
\emph{Drinking tea can aid in digestion, while consuming carbonated beverages may lead to bloating and indigestion.}
&
$\forall x \forall y\,\big(
    \text{DrinkingTea}(x) \land \text{ConsumingCarbonatedBeverages}(y)
    \rightarrow
    (\text{AidsDigestion}(x) \land \lnot \text{CausesBloatingOrIndigestion}(x))
    \land
    (\lnot \text{AidsDigestion}(y) \land \text{CausesBloatingOrIndigestion}(y))
\big)$
\\[0.8em]

\emph{A healthy diet includes fruits, vegetables, and whole grains.}
&
$\forall x\,\big(
    \text{HealthyDiet}(x)
    \rightarrow
    (\text{IncludesFruits}(x) \land \text{IncludesVegetables}(x) \land \text{IncludesWholeGrains}(x))
\big)$
\\[0.8em]

\emph{People who exercise regularly and maintain a balanced diet are generally healthier.}
&
$\forall x\,\big(
    \text{Person}(x) \land \text{RegularExercise}(x) \land \text{BalancedDiet}(x)
    \rightarrow
    \text{GenerallyHealthier}(x)
\big)$
\\
\end{tabular}
\end{table}

\vspace{0.5em}

\begin{block}{Observation}
\begin{itemize}
    \item Each pair is a \textbf{well-formed} NL$\rightarrow$FOL translation.
    \item Across many such pairs, predicate names are chosen \alert{independently}
          (\texttt{HealthyDiet}, \texttt{BalancedDiet}, \texttt{GenerallyHealthier}, \ldots).
    \item NL2UNIFOL aims to turn many such pairs into a theory with a \emph{uniform vocabulary}.
\end{itemize}
\end{block}

\end{frame}

}\label{sec:motivation-&-context} % end background

%===============================================================================
\section[Framework]{NL2UNIFOL Framework}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}[allowframebreaks]{High-Level View of NL2UNIFOL}

%\begin{block}{What do we mean by \emph{uniform}?}
%All formulae in the theory share the same vocabulary, i.e., a common space of predicate and constant names.
%\end{block}

\begin{block}{Pipeline}
\begin{enumerate}
    % (1)
    %independent translation of NL sentences into preliminary FOL formulae, (2) parsing of formulae to extract
    %predicate and constant names, (3) creation of the embedding-based representations of the extracted names, (4)
    %computation of similarity score matrices for predicates and constants, (5) graph clustering to identify sets of
    %names to be merged, and (6) generation of the final uniform FOL theory by replacing the names in the preliminary
    %formulae with the merged ones.
    \item \textbf{Independent NL$\rightarrow$FOL translation}: each NL sentence is translated into a preliminary FOL formula
    \item \textbf{Symbol collection}: parse the formulae to extract predicate and constant names
    \item \textbf{Context-based embeddings}: for each predicate and constant, build a context string describing how it is used in the theory, and feed it into a transformer encoder to obtain a dense embedding
    \item \textbf{Similarity matrices}: compute semantic and syntactic similarity matrices
    \item \textbf{Graph clustering}: build graphs where nodes are symbols and edges represent similarity above a threshold, then identify connected components as clusters of symbols to be merged
    \item \textbf{Reconciliation}: for each cluster, choose a representative name and replace all symbols in the formulae with their representative, yielding a \emph{uniform} FOL theory
\end{enumerate}
\end{block}

\framebreak

\begin{figure}
    \centering
    \includegraphics[width=0.9\textwidth]{imgs/workflow}
    \caption{NL2UNIFOL pipeline overview.}
    \caption{
        The NL2UNIFOL pipeline consists of six main steps: (1) independent translation, (2) symbol collection, (3) context-based embeddings, (4) similarity matrices, (5) graph clustering, and (6) reconciliation to produce a uniform FOL theory.
    }
    \label{fig:nl2unifol-pipeline}
\end{figure}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 1: Independent NL$\rightarrow$FOL Translation}

\begin{block}{Input}
\begin{itemize}
    \item NL sentences in the same domain (here: nutrition)
\end{itemize}
\end{block}

\begin{block}{LLM-based translation}
\begin{itemize}
    \item Use instruction-tuned LLMs
    \item Prompt with:
    \begin{itemize}
        \item FOL \emph{grammar} and allowed operators
        \item naming rules for predicates and variables
        \item output \emph{formatting} instructions
        \item \emph{few-shot examples}
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Output}
\begin{itemize}
    \item One syntactically-checked FOL formula per sentence
    \item Still \alert{non-uniform} symbol space
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 2: Symbol Collection with Logical Constraints}

\begin{block}{Parsing}
\begin{itemize}
    \item Parse each preliminary formula with a grammar checker
    \item Extract:
    \begin{itemize}
        \item predicate symbols as pairs $\langle N, A\rangle$ (name and arity)
        \item constant names
        \item number of occurrences and argument positions
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Logical constraint}
\begin{itemize}
    \item Predicates with different arity \emph{cannot} be merged
    \item Argument positions will be used later to build context-based embeddings
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 3: Context-Based Embeddings}

\begin{block}{Key idea}
Symbols should be represented not only by their surface form, but by \alert{how they are used} in the corpus.
\end{block}

\begin{block}{Predicate context}
\begin{itemize}
    \item For a predicate $N$ of arity $A$, collect all terms seen in each argument position
    \item Verbalise into a short NL description of the form:\\
    \emph{Predicate ‘N’ relates something like ‘T\_{1,1}, ..., T\_{1,n\_1}’ to ... ‘T\_{A,1}, ..., T\_{A,n\_A}’.}
\end{itemize}
\end{block}

\begin{block}{Embeddings}
\begin{itemize}
    \item Feed these context strings into a transformer encoder
    \item Obtain dense, L2-normalised semantic vectors for each symbol
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 4: Similarity Matrices}

\begin{block}{Semantic and syntactic similarity}
For predicates (similarly for constants):
\begin{itemize}
    \item Semantic: cosine similarity between embeddings
          $\Rightarrow M_{p,\text{sem}}$
    \item Syntactic: normalised Levenshtein distance between names
          $\Rightarrow M_{p,\text{syn}}$
\end{itemize}
\end{block}

\begin{block}{Arity mask}
\begin{itemize}
    \item Build $M_a$ where entries for pairs with different arity are set to zero
\end{itemize}
\end{block}

\begin{block}{Final similarity}
\[
M_p = \big((1 - \alpha_p)\, M_{p,\text{sem}} + \alpha_p\, M_{p,\text{syn}}\big) \odot M_a
\]
\[
M_c = (1 - \alpha_c)\, M_{c,\text{sem}} + \alpha_c\, M_{c,\text{syn}}
\]
\begin{itemize}
    \item $\alpha_p, \alpha_c \in [0,1]$ control the weight of syntactic similarity
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 5–6: Graph Clustering and Reconciliation}

\begin{block}{Graph construction}
\begin{itemize}
    \item Build graphs for predicates and constants:
    \begin{itemize}
        \item nodes: symbols
        \item edges: similarity $> \tau_p$ (or $\tau_c$)
    \end{itemize}
    \item Connected components = clusters of symbols to be merged
\end{itemize}
\end{block}

\begin{block}{Representative and reconciliation}
\begin{itemize}
    \item For each cluster, choose as representative the name with \emph{most occurrences}
    \item Replace all symbols in preliminary formulas with their cluster representative
    \item Output: a \alert{uniform} FOL theory
\end{itemize}
\end{block}

\end{frame}

} % end framework

\section[Experiment]{Experimental Setup \& Results}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}[allowframebreaks]{Dataset and NL$\rightarrow$FOL Baseline}

\begin{block}{MALLS dataset}
\begin{itemize}
    \item NL–FOL pairs from multiple domains
    \item We focus on a \alert{nutrition} subset
    \item 748 pairs for evaluation, 27 for few-shot prompting
\end{itemize}
\end{block}

\begin{block}{Models evaluated}
\begin{itemize}
    \item Qwen 2.5 Instruct (7B)
    \item LLaMA 3 Instruct (8B)
    \item Mistral 7B Instruct
    \item Phi 3.1 Instruct (7B)
\end{itemize}
\end{block}

    \framebreak

\begin{figure}
    \centering
    \includegraphics[width=0.9\textwidth]{imgs/syntax_translation_performance}
    \caption{
        Syntactic correctness of the generated FOL formulae across different LLM families and prompting strategies.
    }
    \label{fig:syntax_translation_performance}
    \end{figure}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}[allowframebreaks]{Hyperparameter Selection for Predicate Clustering}

\begin{block}{Hyperparameters}
\begin{itemize}
    \item $\alpha_p$: trade-off between semantic vs syntactic similarity
    \item $\tau_p$: similarity threshold for creating edges in the predicate graph
\end{itemize}
\end{block}

\begin{block}{Grid search}
\begin{itemize}
    \item $\alpha_p \in \{0.0, 0.1, \dots, 1.0\}$
    \item $\tau_p \in \{0.525, 0.55, \dots, 1.0\}$
\end{itemize}
\end{block}

\begin{block}{Selection criterion}
\begin{itemize}
    \item Want many non-trivial clusters \emph{without} over-merging everything
    \item Define a score $S(\alpha_p, \tau_p)$ that:
    \begin{itemize}
        \item rewards the number of non-trivial clusters
        \item penalises excessive merges
        \item prefers stable optima (no sharp peaks)
    \end{itemize}
\end{itemize}
\end{block}

\framebreak

\begin{figure}
    \centering
    \includegraphics[width=0.9\textwidth]{imgs/predicate_merging_plot}
    \caption{
        Predicate clustering behaviour varying $\alpha_p$ (semantic vs syntactic weight) and $\tau_p$ (similarity threshold).
        The heatmap shows the score $S(\alpha_p, \tau_p)$, with brighter colors indicating better configurations.
        The best configuration is $\alpha_p^\star = 0.1$, $\tau_p^\star = 0.9$.
    }
    \label{fig:merging}
\end{figure}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}[allowframebreaks]{Predicate Clustering: Quantitative Results}

\begin{block}{Best configuration}
\begin{itemize}
    \item $\alpha_p^\star = 0.1$ (mostly semantic, slightly syntactic)
    \item $\tau_p^\star = 0.9$ (conservative threshold)
\end{itemize}
\end{block}

\begin{block}{Vocabulary reduction}
\begin{itemize}
    \item \alert{1170} unique predicates in the generated formulas
    \item After merging: \alert{863} unique predicates
    \item $\Rightarrow$ about \alert{26.2\%} reduction
\end{itemize}
\end{block}

\begin{block}{Clusters}
\begin{itemize}
    \item 307 predicates are merged into 140 non-trivial clusters
    \item Remaining predicates are singletons (not merged)
\end{itemize}
\end{block}

    \framebreak

    \begin{figure}
        \centering
        \includegraphics[width=0.9\textwidth]{imgs/predicate_similarity_graph_a1_t900}
        \caption{
            Non-trivial predicate clusters obtained for $\alpha_p = 0.1$ and $\tau_p = 0.9$.
            %
            Bubbles represent the clusters, with size proportional to the number of predicates in the cluster.
            %
            The representative names of big clusters are reported as labels inside the bubbles.
        }
        \label{fig:translation_results}
    \end{figure}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}[allowframebreaks]{Error Taxonomy for Predicate Merges}

\begin{block}{Relation-level taxonomy}
Each mapping (\emph{predicate} $\rightarrow$ \emph{representative}) is classified as:
\begin{itemize}
    \item \textbf{OK}: valid mapping, representative is a good semantic generalisation
    \item \textbf{Egen}: over-generalisation (too abstract)
    \item \textbf{Espe}: over-specification (too specific)
    \item \textbf{Ecomp}: compositional reduction (only part of the meaning is preserved)
    \item \textbf{Esem}: semantic mismatch (different concepts)
    \item \textbf{Epol}: polarity inversion (opposite meaning)
    \item \textbf{Ecat}: ontological shift (different ontological type)
    \item \textbf{Ehall}: unsupported generalisation (spurious abstraction)
\end{itemize}
\end{block}

\begin{block}{Annotation}
\begin{itemize}
    \item Three human annotators + majority voting with severity-based tie-breaking
    \item Also compare with automatic annotation from ChatGPT
\end{itemize}
\end{block}

    \allowbreak

\begin{table}[t]
\centering

\midrule

\label{tab:error_analysis}
\end{table}


\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Error Distribution: How Good Are the Merges?}

\begin{block}{Overall picture (majority vote)}
\begin{itemize}
    \item Around \alert{46\%} of mappings are \textbf{OK}
    \item Most common error: \alert{over-generalisation} (\textbf{Egen})
    \item Many errors are relatively mild:
    \begin{itemize}
        \item \textbf{Egen}, \textbf{Espe}, \textbf{Ecomp}, \textbf{Ehall} can be acceptable depending on the required level of abstraction
    \end{itemize}
    \item More severe errors:
    \begin{itemize}
        \item \textbf{Esem}, \textbf{Ecat}, \textbf{Epol} account for about a quarter of the mappings
        \item Polarity inversions (\textbf{Epol}) are rare (around 1–2\%)
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Takeaway}
\begin{itemize}
    \item NL2UNIFOL can achieve substantial vocabulary unification
    \item But symbol clustering must be treated as a \emph{noisy step},
          with error types that depend on the application’s tolerance
\end{itemize}
\end{block}

\end{frame}

\section[Discussion]{Discussion \& Outlook}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}{What NL2UNIFOL Provides}

\begin{block}{Strengths}
\begin{itemize}
    \item End-to-end pipeline from NL sentences to a \emph{uniform} FOL theory
    \item Embedding-based symbol representations that use \emph{logical context of use}
    \item Combination of:
    \begin{itemize}
        \item semantic similarity
        \item syntactic similarity
        \item logical constraints (arity, occurrence-based representative)
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Limitations}
\begin{itemize}
    \item Merging is imperfect: non-negligible portion of semantically problematic merges
    \item Current evaluation is limited to one domain (nutrition) and relatively small models
    \item Clustering is not yet integrated with downstream reasoning as a feedback signal
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Future Work}

\begin{block}{Improving symbol clustering}
\begin{itemize}
    \item Integrate LLMs more tightly into the clustering step
    \item Use logical reasoning (e.g., proof attempts, consistency checks) to veto bad merges
    \item Explore alternative similarity functions and clustering algorithms
\end{itemize}
\end{block}

\begin{block}{Broadening the scope}
\begin{itemize}
    \item Apply NL2UNIFOL to:
    \begin{itemize}
        \item legal corpora and regulatory text
        \item medical guidelines and clinical protocols
        \item richer ethical domains
    \end{itemize}
    \item Extend to other logics (e.g., \alert{modal/deontic logics}) for normative reasoning
\end{itemize}
\end{block}

\end{frame}

} % end discussion
