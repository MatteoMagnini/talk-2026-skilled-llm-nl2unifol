\begin{frame}{Contribution}

\begin{block}{Main contributions of our work}
    \begin{itemize}
        \item Introduce \textbf{NL2UNIFOL}, a framework that turns NL sentences into a \alert{uniform} FOL theory
        \item Propose a \textbf{symbol clustering} method that safely merges predicate and constant names
        \item Combine \textbf{semantic} and \textbf{syntactic} similarity under logical constraints
        \item Empirically study the quality of the merges on a \textbf{nutrition} subset of MALLS
    \end{itemize}
\end{block}

\section[Background]{Motivation \& Context}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}{Motivation: Norms from Heterogeneous Stakeholders}

\begin{block}{Sources of normative text}
\begin{itemize}
    \item Official regulations (e.g., GDPR, MiFID II), guidelines, institutional policies
    \item Individual or group descriptions of moral values and principles
\end{itemize}
\end{block}

\begin{block}{Vocabulary misalignment}
\begin{itemize}
    \item Different words for the same entity: \emph{customer} vs \emph{client}
    \item Different phrasings for the same relation: \emph{is employed by} vs \emph{works for}
    \item Typos, singular/plural, synonyms, hypernyms, paraphrases
\end{itemize}
\end{block}

\begin{block}{Consequence}
\begin{itemize}
    \item If every sentence is translated independently into FOL with its own vocabulary,
          a reasoning engine cannot easily relate different norms and derive meaningful conclusions.
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Use Case: DJ4ME and Machine Ethics}

\begin{block}{DJ4ME (Dialogue Jiminy)}
\begin{itemize}
    \item Architecture for machine ethics based on persuasion dialogues among stakeholders' avatars
    \item Needs:
    \begin{itemize}
        \item \textbf{Norm mining} from NL corpora
        \item Explanations of moral decisions in terms of extracted norms
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Why uniform FOL matters here}
\begin{itemize}
    \item Norms must be represented in a \textbf{shared logical vocabulary}
    \item Otherwise:
    \begin{itemize}
        \item stakeholders talk “past each other” at the symbolic level
        \item reasoning and explanation across norms become fragile or impossible
    \end{itemize}
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Limit of Existing NL$\rightarrow$FOL Approaches}

\begin{block}{Recent progress}
\begin{itemize}
    \item LLM-based methods for NL$\rightarrow$FOL translation
    \item Benchmarks (e.g., MALLS, FOLIO) and evaluation metrics
    \item Strong results on \textbf{single-sentence} translation
\end{itemize}
\end{block}

\begin{block}{Missing piece}
\begin{itemize}
    \item Existing works focus on isolated sentences
    \item Predicate naming, constants, and variables are decided \textbf{locally}
    \item No explicit mechanism for:
    \begin{itemize}
        \item symbol reuse
        \item predicate alignment
        \item theory-level coherence across a corpus
    \end{itemize}
\end{itemize}
\end{block}

\end{frame}

}\label{sec:motivation-&-context} % end background

%===============================================================================
\section[Framework]{NL2UNIFOL Framework}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}{High-Level View of NL2UNIFOL}

\begin{block}{Definition of “uniform”}
All formulae in the theory share the same vocabulary, i.e., a common space of predicate and constant names.
\end{block}

\begin{block}{Pipeline}
\begin{enumerate}
    \item \textbf{Independent translation}:
          use an LLM to translate each NL sentence into a preliminary FOL formula
    \item \textbf{Symbol clustering}:
          identify and safely merge predicate and constant names
    \item \textbf{Reconciliation}:
          replace original names with cluster representatives to obtain the final uniform theory
\end{enumerate}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 1: Independent NL$\rightarrow$FOL Translation}

\begin{block}{Input}
\begin{itemize}
    \item NL sentences in the same domain (here: nutrition)
\end{itemize}
\end{block}

\begin{block}{LLM-based translation}
\begin{itemize}
    \item Use instruction-tuned LLMs (e.g., Qwen 2.5, LLaMA 3, Mistral, Phi 3.1)
    \item Prompt with:
    \begin{itemize}
        \item explicit FOL grammar and allowed operators
        \item naming rules for predicates and variables
        \item no functions, only predicates and (when needed) constants
        \item few-shot examples (27 pairs from MALLS nutrition)
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Output}
\begin{itemize}
    \item One syntactically-checked FOL formula per sentence
    \item Still \textbf{non-uniform} symbol space
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 2: Symbol Collection with Logical Constraints}

\begin{block}{Parsing}
\begin{itemize}
    \item Parse each preliminary formula with a CFG (implemented via Lark)
    \item Extract:
    \begin{itemize}
        \item predicate symbols as pairs $\langle N, A\rangle$ (name and arity)
        \item constant names
        \item number of occurrences and argument positions
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Logical constraint}
\begin{itemize}
    \item Predicates with different arity \textbf{cannot} be merged
    \item Argument positions will be used later to build context-based embeddings
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 3: Context-Based Embeddings}

\begin{block}{Key idea}
Symbols should be represented not only by their surface form, but by \textbf{how they are used}
in the preliminary theory.
\end{block}

\begin{block}{Predicate context}
\begin{itemize}
    \item For a predicate $N$ of arity $A$, collect all terms seen in each argument position
    \item Verbalise into a short NL description of the form:\\
    \emph{Predicate ‘N’ relates something like ‘T\_{1,1}, ..., T\_{1,n\_1}’ to ... ‘T\_{A,1}, ..., T\_{A,n\_A}’.}
\end{itemize}
\end{block}

\begin{block}{Constant context}
\begin{itemize}
    \item For a constant $c$, collect predicates and argument positions in which it occurs
    \item Verbalise as:\\
    \emph{Constant ‘c’ is used as argument i of predicates ‘P\_1, ..., P\_m’, as argument j of predicates ‘P\_1, ..., P\_n’.}
\end{itemize}
\end{block}

\begin{block}{Embeddings}
\begin{itemize}
    \item Feed these context strings into a transformer encoder (e.g., all-MiniLM-L6-v2)
    \item Obtain dense, L2-normalised semantic vectors for each symbol
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 4: Similarity Matrices}

\begin{block}{Semantic and syntactic similarity}
For predicates (similarly for constants):
\begin{itemize}
    \item Semantic: cosine similarity between embeddings
          $\Rightarrow M_{p,\text{sem}}$
    \item Syntactic: normalised Levenshtein distance between names
          $\Rightarrow M_{p,\text{syn}}$
\end{itemize}
\end{block}

\begin{block}{Arity mask}
\begin{itemize}
    \item Build $M_a$ where entries for pairs with different arity are set to zero
\end{itemize}
\end{block}

\begin{block}{Final similarity}
\[
M_p = \big((1 - \alpha_p)\, M_{p,\text{sem}} + \alpha_p\, M_{p,\text{syn}}\big) \odot M_a
\]
\[
M_c = (1 - \alpha_c)\, M_{c,\text{sem}} + \alpha_c\, M_{c,\text{syn}}
\]
\begin{itemize}
    \item $\alpha_p, \alpha_c \in [0,1]$ control the weight of syntactic similarity
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Step 5–6: Graph Clustering and Reconciliation}

\begin{block}{Graph construction}
\begin{itemize}
    \item Build graphs for predicates and constants:
    \begin{itemize}
        \item nodes: symbols
        \item edges: similarity $> \tau_p$ (or $\tau_c$)
    \end{itemize}
    \item Connected components = clusters of symbols to be merged
\end{itemize}
\end{block}

\begin{block}{Representative and reconciliation}
\begin{itemize}
    \item For each cluster, choose as representative the name with \textbf{most occurrences}
    \item Replace all symbols in preliminary formulas with their cluster representative
    \item Output: a \textbf{uniform} FOL theory
\end{itemize}
\end{block}

\end{frame}

} % end framework

\section[Experiment]{Experimental Setup \& Results}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}{Dataset and NL$\rightarrow$FOL Baseline}

\begin{block}{MALLS dataset}
\begin{itemize}
    \item NL–FOL pairs from multiple domains
    \item We focus on a \textbf{nutrition} subset (food recommendation)
    \item 748 pairs for evaluation, 27 for few-shot prompting
\end{itemize}
\end{block}

\begin{block}{Models evaluated}
\begin{itemize}
    \item Qwen 2.5 Instruct (7B)
    \item LLaMA 3 Instruct (8B)
    \item Mistral 7B Instruct
    \item Phi 3.1 Instruct (7B)
\end{itemize}
\end{block}

\begin{block}{Syntactic correctness}
\begin{itemize}
    \item Metric: percentage of generated formulas that are valid under our CFG
    \item Few-shot prompting improves all models
    \item Qwen 2.5 Instruct (few-shot) gives the best syntactic correctness
          $\Rightarrow$ used as base for clustering experiments
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Hyperparameter Selection for Predicate Clustering}

\begin{block}{Hyperparameters}
\begin{itemize}
    \item $\alpha_p$: trade-off between semantic vs syntactic similarity
    \item $\tau_p$: similarity threshold for creating edges in the predicate graph
\end{itemize}
\end{block}

\begin{block}{Grid search}
\begin{itemize}
    \item $\alpha_p \in \{0.0, 0.1, \dots, 1.0\}$
    \item $\tau_p \in \{0.525, 0.55, \dots, 1.0\}$
\end{itemize}
\end{block}

\begin{block}{Selection criterion}
\begin{itemize}
    \item Want many non-trivial clusters \textbf{without} over-merging everything
    \item Define a score $S(\alpha_p, \tau_p)$ that:
    \begin{itemize}
        \item rewards the number of non-trivial clusters
        \item penalises excessive merges
        \item prefers stable optima (no sharp peaks)
    \end{itemize}
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Predicate Clustering: Quantitative Results}

\begin{block}{Best configuration}
\begin{itemize}
    \item $\alpha_p^\star = 0.1$ (mostly semantic, slightly syntactic)
    \item $\tau_p^\star = 0.9$ (conservative threshold)
\end{itemize}
\end{block}

\begin{block}{Vocabulary reduction}
\begin{itemize}
    \item 1170 unique predicates in the generated formulas
    \item After merging: 863 unique predicates
    \item $\Rightarrow$ about \textbf{26.2\%} reduction
\end{itemize}
\end{block}

\begin{block}{Clusters}
\begin{itemize}
    \item 307 predicates are merged into 140 non-trivial clusters
    \item Remaining predicates are singletons (not merged)
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Error Taxonomy for Predicate Merges}

\begin{block}{Relation-level taxonomy}
Each mapping (\emph{predicate} $\rightarrow$ \emph{representative}) is classified as:
\begin{itemize}
    \item \textbf{OK}: valid mapping, representative is a good semantic generalisation
    \item \textbf{Egen}: over-generalisation (too abstract)
    \item \textbf{Espe}: over-specification (too specific)
    \item \textbf{Ecomp}: compositional reduction (only part of the meaning is preserved)
    \item \textbf{Esem}: semantic mismatch (different concepts)
    \item \textbf{Epol}: polarity inversion (opposite meaning)
    \item \textbf{Ecat}: ontological shift (different ontological type)
    \item \textbf{Ehall}: unsupported generalisation (spurious abstraction)
\end{itemize}
\end{block}

\begin{block}{Annotation}
\begin{itemize}
    \item Three human annotators + majority voting with severity-based tie-breaking
    \item Also compare with automatic annotation from ChatGPT
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Error Distribution: How Good Are the Merges?}

\begin{block}{Overall picture (majority vote)}
\begin{itemize}
    \item Around \textbf{46\%} of mappings are \textbf{OK}
    \item Most common error: \textbf{over-generalisation} (Egen)
    \item Many errors are relatively mild:
    \begin{itemize}
        \item Egen, Espe, Ecomp, Ehall can be acceptable depending on the required level of abstraction
    \end{itemize}
    \item More severe errors:
    \begin{itemize}
        \item Esem, Ecat, Epol account for about a quarter of the mappings
        \item Polarity inversions (Epol) are rare (around 1–2\%)
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Takeaway}
\begin{itemize}
    \item NL2UNIFOL can achieve substantial vocabulary unification
    \item But symbol clustering must be treated as a \textbf{noisy step},
          with error types that depend on the application’s tolerance
\end{itemize}
\end{block}

\end{frame}

} % end experiment

\section[Discussion]{Discussion \& Outlook}
%===============================================================================

{
\usebackgroundtemplate{}

%-------------------------------------------------------------------------------
\begin{frame}{What NL2UNIFOL Provides}

\begin{block}{Strengths}
\begin{itemize}
    \item End-to-end pipeline from NL sentences to a \textbf{uniform} FOL theory
    \item Embedding-based symbol representations that use \textbf{logical context of use}
    \item Combination of:
    \begin{itemize}
        \item semantic similarity
        \item syntactic similarity
        \item logical constraints (arity, occurrence-based representative)
    \end{itemize}
\end{itemize}
\end{block}

\begin{block}{Limitations}
\begin{itemize}
    \item Merging is imperfect: non-negligible portion of semantically problematic merges
    \item Current evaluation is limited to one domain (nutrition) and relatively small models
    \item Clustering is not yet integrated with downstream reasoning as a feedback signal
\end{itemize}
\end{block}

\end{frame}

%-------------------------------------------------------------------------------
\begin{frame}{Future Work}

\begin{block}{Improving symbol clustering}
\begin{itemize}
    \item Integrate LLMs more tightly into the clustering step
    \item Use logical reasoning (e.g., proof attempts, consistency checks) to veto bad merges
    \item Explore alternative similarity functions and clustering algorithms
\end{itemize}
\end{block}

\begin{block}{Broadening the scope}
\begin{itemize}
    \item Apply NL2UNIFOL to:
    \begin{itemize}
        \item legal corpora and regulatory text
        \item medical guidelines and clinical protocols
        \item richer ethical domains in DJ4ME
    \end{itemize}
    \item Extend to other logics (e.g., modal/deontic logics) for normative reasoning
\end{itemize}
\end{block}

\end{frame}

} % end discussion

%===============================================================================
\begin{frame}{Conclusions}

\begin{block}{In a nutshell}
\begin{itemize}
    \item We introduce NL2UNIFOL, targeting \textbf{uniform FOL theories} from NL corpora
    \item The core is a context-aware symbol clustering mechanism combining semantic and syntactic information under logical constraints
    \item Experiments on MALLS (nutrition domain) show:
    \begin{itemize}
        \item substantial vocabulary reduction
        \item a mix of correct and noisy merges, with an explicit error taxonomy
    \end{itemize}
\end{itemize}
\end{block}
